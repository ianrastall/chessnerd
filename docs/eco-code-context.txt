            const third = document.getElementById('eco3');
            const content = document.getElementById('ecoContent');

            document.getElementById('backButton').addEventListener('click', () => {
                window.location.href = 'index.html';
            });

            function populate(select, codes) {
                select.innerHTML = '<option value="">- Select -</option>';

                codes.forEach(code => {
                    const name = ecoData[code]?.name || code;
                    const option = document.createElement('option');
                    option.value = code;
                    option.textContent = `${code}: ${name}`;
                    select.appendChild(option);
                });
            }

            function render(code) {
                if (!code || !ecoData[code]) {
                    content.innerHTML = '<p style="color: var(--text-secondary)">Select a code to view details.</p>';
                    return;
                }

                const contentText = typeof ecoData[code].content === 'string' ? ecoData[code].content : '';
                let html = `<p>${contentText.replace(/\n/g, '<br>')}</p>`;

                if (ecoData[code].openings && ecoData[code].openings.length > 0) {
                    html += '<h5>Named Openings</h5>';
                    ecoData[code].openings.forEach(op => {
                        const openingText = typeof op === 'string' ? op : String(op ?? '');
                        html += `<div class="eco-move-text">${openingText.replace(/\n/g, '<br>')}</div>`;
                    });
                }
                content.innerHTML = html;
            }

            first.addEventListener('change', () => {
                const c = first.value;
                second.disabled = true;
                third.disabled = true;
                second.innerHTML = '';
                third.innerHTML = '';
                
                render(c);

                if (c && ecoData[c]?.children?.length) {
                    populate(second, ecoData[c].children);
                    second.disabled = false;
                }
            });

            second.addEventListener('change', () => {
                const c = second.value;
                third.disabled = true;
                third.innerHTML = '';
                
                render(c);

                if (c && ecoData[c]?.children?.length) {
                    populate(third, ecoData[c].children);
                    third.disabled = false;
                }
            });

            third.addEventListener('change', () => {
                render(third.value);
            });

            async function init() {
                try {
                    const response = await fetch('js/eco-code.json');
                    if (!response.ok) throw new Error('Failed to load eco-code.json');
                    
                    ecoData = await response.json();
                    
                    populate(first, ['A', 'B', 'C', 'D', 'E']);
                    content.innerHTML = '<p style="color: var(--text-secondary)">Please select a volume to begin.</p>';
                } catch (error) {
                    console.error(error);
                    content.innerHTML = `<div class="error">Error: Could not load ECO data. Make sure 'js/eco-code.json' exists.</div>`;
                    first.innerHTML = '<option value="">Error loading data</option>';
                    first.disabled = true;
                    second.innerHTML = '<option value="">Unavailable</option>';
                    second.disabled = true;
                    third.innerHTML = '<option value="">Unavailable</option>';
                    third.disabled = true;
                }
            }

            init();
        });
